find_package(SCIM)
if(SCIM_FOUND)
    include_directories(${SCIM_INCLUDE_DIR})
    set(SCIM_MODULE_DIR "${PC_SCIM_PREFIX}/lib/scim-1.0/1.4.0")
    set(kimtoy_scim_kconfig_SRCS scim_kconfig_config.cpp)
    add_library(kimtoy-scim-kconfig MODULE ${kimtoy_scim_kconfig_SRCS})
    target_link_libraries(kimtoy-scim-kconfig ${KDE4_KDECORE_LIBS} ${SCIM_LIBRARIES})
    set_target_properties(kimtoy-scim-kconfig PROPERTIES OUTPUT_NAME kconfig4 PREFIX "")
    install(TARGETS kimtoy-scim-kconfig DESTINATION ${SCIM_MODULE_DIR}/Config)
endif(SCIM_FOUND)
