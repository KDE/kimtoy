find_package(SCIM)
if(SCIM_FOUND)
    include_directories(${SCIM_INCLUDE_DIR})
    set(SCIM_LIBEXEC_DIR "${PC_SCIM_PREFIX}/lib/scim-1.0")
    set(SCIM_ICON_DIR "${PC_SCIM_PREFIX}/share/scim/icons")
    configure_file(config-scim.h.cmake config-scim.h)
    set(kimtoy_scim_panel_SRCS main.cpp)
    qt4_generate_moc(${kimtoy_scim_panel_SRCS} ${CMAKE_CURRENT_BINARY_DIR}/main.moc)
    add_executable(kimtoy-scim-panel ${kimtoy_scim_panel_SRCS})
    target_link_libraries(kimtoy-scim-panel ${QT_QTCORE_LIBRARY} ${QT_QTDBUS_LIBRARY} ${SCIM_LIBRARIES})
    set_target_properties(kimtoy-scim-panel PROPERTIES OUTPUT_NAME scim-panel-impanel)
    install(TARGETS kimtoy-scim-panel DESTINATION ${SCIM_LIBEXEC_DIR})
endif(SCIM_FOUND)
